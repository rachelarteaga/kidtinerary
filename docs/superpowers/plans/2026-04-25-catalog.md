# Catalog Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship the Catalog defined in [docs/superpowers/specs/2026-04-25-catalog-design.md](docs/superpowers/specs/2026-04-25-catalog-design.md): replace the placeholder `/catalog` page with a real personal master library — list, filters, sorts, two intake CTAs (Add and Help-me-find), kid attribution, and an LLM web-search slide-over.

**Architecture:** One bundle, twelve phases. Phase 1 ships a previewable catalog list against existing `user_activity` data so the user can see something running early; Phase 2 lands schema; Phases 3–7 layer in attribution, filters, sorts, and the detail-drawer extension; Phases 8–10 build the Add/Help-me-find surfaces; Phases 11–12 do mobile and cleanup.

**Tech Stack:** Next.js 16 (App Router), React 19, Supabase SSR (`@supabase/ssr`), Tailwind v4 (existing tokens in `globals.css`), Vitest + Testing Library (logic-only tests; UI verified visually). LLM via Vercel AI Gateway through `ai` SDK + provider/model strings (per Vercel knowledge update).

**Branch:** `catalog` (created in Phase 0).

**Note on Next.js:** Read `node_modules/next/dist/docs/` before reaching for App Router APIs from memory. There are breaking changes from older training data.

**Test philosophy:** TDD for pure logic (filter URL serialization, season bucketing, source-filter mapping, kid-tag derivation). Server actions wrapping Supabase rely on type safety + manual verification unless they have non-trivial branching. UI components are verified visually via the dev preview at `/catalog` (the existing pattern in `tests/` is logic-only — no React component tests in this codebase).

**Preview checkpoints:**
- **End of Phase 1** — `/catalog` renders real `user_activity` rows. Minimal row treatment, no filters yet.
- **End of Phase 9** — full UI complete with mock LLM results in Help-me-find.
- **End of Phase 10** — Help-me-find calls real LLM.

**Files:**
- Create: `supabase/migrations/034_catalog_auto_tag_trigger.sql`

- [ ] **Step 1: Write the trigger**

```sql
-- 034_catalog_auto_tag_trigger.sql
create or replace function append_kid_tag_on_planner_entry()
returns trigger
language plpgsql
security definer
set search_path = public
as $$
begin
  -- planner_entry has user_activity_id and kid_id (verify column names against
  -- migrations 010 + 013 + 015 if this fails — adjust to actual schema).
  update user_activity
  set kid_tags = (
    case when new.kid_id is null then kid_tags
         when new.kid_id = any(kid_tags) then kid_tags
         else array_append(kid_tags, new.kid_id)
    end
  )
  where id = new.user_activity_id;
  return new;
end;
$$;

create trigger trg_planner_entry_auto_tag_kid
  after insert on planner_entry
  for each row
  execute function append_kid_tag_on_planner_entry();
```

```sql
update user_activity ua

from (
  select user_activity_id, array_agg(distinct kid_id) as kids

  where kid_id is not null
```

```ts
export async function tagKidOnUserActivity(userActivityId: string, kidId: string) {
  const supabase = await createServerClient();
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) return { error: "Not signed in." };

  const { error } = await supabase.rpc("array_append_unique_kid_tag", {
    user_activity_id: userActivityId,
    kid_id: kidId,
  });
  if (error) return { error: error.message };
  return { error: null };
}

export async function untagKidOnUserActivity(userActivityId: string, kidId: string) {
  const supabase = await createServerClient();
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) return { error: "Not signed in." };

  // array_remove is built-in
  const { error } = await supabase.rpc("array_remove_kid_tag", {
    user_activity_id: userActivityId,
    kid_id: kidId,
  });
  if (error) return { error: error.message };
```

```sql
-- 036_catalog_kid_tag_rpcs.sql
create or replace function array_append_unique_kid_tag(user_activity_id uuid, kid_id uuid)
returns void
language sql
security definer
set search_path = public
as $$
  update user_activity
  set kid_tags = case
    when kid_id = any(kid_tags) then kid_tags
    else array_append(kid_tags, kid_id)
  end
  where id = user_activity_id
    and exists (
      select 1 from user_activity ua
      where ua.id = user_activity_id and ua.user_id = auth.uid()
    );
$$;

create or replace function array_remove_kid_tag(user_activity_id uuid, kid_id uuid)
returns void
language sql
security definer
set search_path = public
as $$
  update user_activity
  set kid_tags = array_remove(kid_tags, kid_id)
  where id = user_activity_id
    and exists (
      select 1 from user_activity ua
      where ua.id = user_activity_id and ua.user_id = auth.uid()
```

```ts
export async function saveHelpMeFindResult(payload: HelpMeFindResultPayload) {
  const supabase = await createServerClient();
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) return { error: "Not signed in.", userActivityId: null };

  // 1. Insert activity row (origin = 'llm')
  const { data: activityRow, error: aErr } = await supabase
    .from("activities")
    .insert({
      name: payload.name,
      registration_url: payload.url,
      description: payload.description,
      categories: payload.categories,
      age_min: payload.ageMin,
      age_max: payload.ageMax,
      registration_end_date: payload.registrationEndDate,
      origin: "llm",
      // organization handled below
    })
    .select("id")
    .single();
  if (aErr || !activityRow) return { error: aErr?.message ?? "Could not save activity.", userActivityId: null };

  // 2. Insert user_activity (source = 'llm', discovery_query)
  const { data: ua, error: uaErr } = await supabase
    .from("user_activity")
    .insert({
      user_id: user.id,
      activity_id: activityRow.id,
      source: "llm",
      discovery_query: payload.discoveryQuery,
      color: nextUnusedPaletteColor(),  // existing helper
    })
    .select("id")
    .single();
  if (uaErr || !ua) return { error: uaErr?.message ?? "Could not save catalog row.", userActivityId: null };

  return { error: null, userActivityId: ua.id };
}
```

```ts
// src/lib/catalog-filters.ts
export interface FilterState {
  kidIds?: string[];
  source?: "me" | "friends";
  types?: Array<"camp" | "class" | "lesson" | "sport">;
  seasons?: Array<"this-summer" | "this-school-year" | "past" | "unknown">;
  categories?: string[];
}

export function serializeFilterState(state: FilterState): URLSearchParams {
  const p = new URLSearchParams();
  if (state.kidIds?.length) p.set("kids", state.kidIds.join(","));
  if (state.source) p.set("source", state.source);
  if (state.types?.length) p.set("type", state.types.join(","));
  if (state.seasons?.length) p.set("season", state.seasons.join(","));
  if (state.categories?.length) p.set("category", state.categories.join(","));
  return p;
}

export function parseFilterState(params: URLSearchParams): FilterState {
  const out: FilterState = {};
  const kids = params.get("kids");
  if (kids) out.kidIds = kids.split(",");
  const source = params.get("source");
  if (source === "me" || source === "friends") out.source = source;
  const t = params.get("type");
  if (t) out.types = t.split(",") as FilterState["types"];
  const s = params.get("season");
  if (s) out.seasons = s.split(",") as FilterState["seasons"];
  const c = params.get("category");
  if (c) out.categories = c.split(",");
  return out;
}

export function matchesSourceFilter(
  rowSource: "self" | "friend" | "llm",
  filter: "me" | "friends" | undefined
): boolean {
  if (!filter) return true;
  if (filter === "me") return rowSource === "self" || rowSource === "llm";
  return rowSource === "friend";
}

export function bucketSeason(
  startDate: string | null,
  today: Date = new Date()
): "this-summer" | "this-school-year" | "past" | "unknown" {
  if (!startDate) return "unknown";
  const d = new Date(startDate + "T00:00:00Z");
  if (d < today) return "past";
  const month = d.getUTCMonth(); // 0 = Jan
  if (month >= 5 && month <= 7) return "this-summer";  // Jun-Aug
  return "this-school-year";
}
```

```ts
  const body = await req.json();
  const { prompt, context } = body as {
    prompt: string;
    context: {
      kids: Array<{ name: string; ageYears: number; interests: string[] }>;
      address: string | null;
    } | null;
  };

  const systemPrompt = buildSystemPrompt(context);

  const { object } = await generateObject({
    model: "anthropic/claude-sonnet-4-6",

    system: systemPrompt,
    prompt,
    // tools: web search — see Task 10.3
  });

  return NextResponse.json(object);

function buildSystemPrompt(context: PostBody["context"]): string {
  const lines: string[] = [
    "You are a kids' activity finder. Given a parent's request, find 3-5 real activities (camps, classes, lessons, sports) on the open web that fit.",
    "Return structured JSON only. Each result must include a real registration URL.",
    "Be conservative: only return results you can verify are real, currently active programs.",
  ];
  if (context?.address) {
    lines.push(`The user's location: ${context.address}.`);
  }
  if (context?.kids?.length) {
    lines.push("The user's kids:");
    for (const kid of context.kids) {
      lines.push(`- ${kid.name}, age ${kid.ageYears}${kid.interests.length ? `, interests: ${kid.interests.join(", ")}` : ""}.`);
    }
  }
  return lines.join("\n");
}
```
